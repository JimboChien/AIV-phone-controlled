# Setting Enveriment
1. Download

 $ git clone

2. update

 $ sudo apt-get update

# Webcam 
1. Setup

 $ ./mjpg_streamer_setup.sh

2. Run

 $ ./webcam_sh.up

3. To URL localhost:8090

# Android 
1. Install Android IDE

 $ sudo apt-get install arduino

- Control : control_servo
- Combine : avoid_obstacles

 $ ./webcam_sh.up

 $ python3 phone_control_server
